YJ_count_all_factory = {
	every_country = {
		if = {
			limit = {
				has_country_flag = YJ_axis
			}
			add_to_variable = {
				var = global.YJ_axis_all_factory
				value = num_of_civilian_factories
			}
			add_to_variable = {
				var = global.YJ_axis_all_factory
				value = num_of_military_factories
			}
			add_to_variable = {
				var = global.YJ_axis_all_factory
				value = num_of_naval_factories
			}
			add_to_variable = {
				var = global.YJ_axis_arms_factory
				value = num_of_military_factories
			}
			add_to_variable = {
				var = global.YJ_axis_industrial_complex
				value = num_of_civilian_factories
			}
			add_to_variable = {
				var = global.YJ_axis_dockyard
				value = num_of_naval_factories
			}
		}

		if = {
			limit = {
				has_country_flag = YJ_allie
			}
			add_to_variable = {
				var = global.YJ_allie_all_factory
				value = num_of_civilian_factories
			}
			add_to_variable = {
				var = global.YJ_allie_all_factory
				value = num_of_military_factories
			}
			add_to_variable = {
				var = global.YJ_allie_all_factory
				value = num_of_naval_factories
			}
			add_to_variable = {
				var = global.YJ_allie_arms_factory
				value = num_of_military_factories
			}
			add_to_variable = {
				var = global.YJ_allie_industrial_complex
				value = num_of_civilian_factories
			}
			add_to_variable = {
				var = global.YJ_allie_dockyard
				value = num_of_naval_factories
			}
		}
	}
}

YJ_calculate_factory_balance_effect = {
	if = {
        limit = {
            date < 1938.6.1
        }

        set_variable = { global.YJ_axis_advantage = 0 }
        set_variable = { global.YJ_allie_advantage = 0 }
    }

	else = {
    	set_variable = { global.YJ_axis_advantage = global.YJ_allie_all_factory }
    	subtract_from_variable = { var = global.YJ_axis_advantage value = global.YJ_axis_all_factory }

    	if = {
       	 limit = { check_variable = { global.YJ_axis_advantage < 0 } }
       	 set_variable = { global.YJ_axis_advantage = 0 }
    	}

		divide_variable = { var = global.YJ_axis_advantage value = 500 }

			if = {
			limit = {
				check_variable = {
					global.YJ_axis_advantage > 0.5
				}
			}
			set_variable = {
				global.YJ_axis_advantage = 0.5
			}
		}

		set_variable = { global.YJ_allie_advantage = global.YJ_axis_all_factory }
		subtract_from_variable = { var = global.YJ_allie_advantage value = global.YJ_allie_all_factory }

		if = {
			limit = { check_variable = { global.YJ_allie_advantage < 0 } }
			set_variable = { global.YJ_allie_advantage = 0 }
		}

		divide_variable = { var = global.YJ_allie_advantage value = 500 }

			if = {
			limit = {
				check_variable = {
					global.YJ_allie_advantage > 0.5
				}
			}
			set_variable = {
				global.YJ_allie_advantage = 0.5
			}
		}
	}
}

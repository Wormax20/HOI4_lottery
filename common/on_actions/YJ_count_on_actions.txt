on_actions = {
    on_weekly = {
        effect = {
            every_country = {
                if = {
                    limit = {
                        OR = {
                            tag = GER
                            is_in_faction_with = GER
                        }
                    }
                    clr_country_flag = YJ_allie
                    set_country_flag = YJ_axis
                }
                if = {
                    limit = {
                        OR = {
                            tag = SOV
                            is_in_faction_with = SOV
                        }
                    }
                    clr_country_flag = YJ_axis
                    set_country_flag = YJ_allie
                }

                if = {
                    limit = {
                        has_country_flag = YJ_allie
                    }
                    if = {
                        limit = {
                            has_dynamic_modifier = YJ_AXIS_BAL_ideas
                        }
                        remove_dynamic_modifier = {
                            modifier = YJ_AXIS_BAL_ideas
                        }
                    }
                    add_dynamic_modifier = {
                        modifier = YJ_ALLIE_BAL_ideas
                    }
                }
                if = {
                    limit = {
                        has_country_flag = YJ_axis
                    }
                    if = {
                        limit = {
                            has_dynamic_modifier = YJ_ALLIE_BAL_ideas
                        }
                        remove_dynamic_modifier = {
                            modifier = YJ_ALLIE_BAL_ideas
                        }
                    }
                    add_dynamic_modifier = {
                        modifier = YJ_AXIS_BAL_ideas
                    }
                }
            }

            set_variable = { var = global.YJ_axis_all_factory          value = 0 }
            set_variable = { var = global.YJ_axis_arms_factory         value = 0 }
            set_variable = { var = global.YJ_axis_industrial_complex   value = 0 }
            set_variable = { var = global.YJ_axis_dockyard             value = 0 }
            set_variable = { var = global.YJ_allie_all_factory         value = 0 }
            set_variable = { var = global.YJ_allie_arms_factory        value = 0 }
            set_variable = { var = global.YJ_allie_industrial_complex  value = 0 }
            set_variable = { var = global.YJ_allie_dockyard            value = 0 }
            YJ_count_all_factory = yes
            YJ_calculate_factory_balance_effect = yes
        }
    }

    on_daily = {
        effect = {
            set_variable = {
                var = YJ_coal_num
                value = resource@coal
            }
            divide_variable = {
                var = YJ_coal_num
                value = 100
            }
            set_variable = {
                var = YJ_coal_num_0_01
                value = YJ_coal_num
            }
            divide_variable = {
                var = YJ_coal_num_0_01
                value = 100
            }
        }
    }
}

on_actions = {
	on_startup = {
		effect = {
			for_each_scope_loop = {
				array = global.players
				set_country_flag = show_welcome_splash
			}
			random_country = {
				limit = {
					is_ai = no
				}
				set_country_flag = YJ_controller
			}

			every_country = {
				if = {
					limit = {
						OR = {
							tag = USA
							tag = ENG
							tag = FRA
							tag = GER
							tag = ITA
							tag = SOV
							tag = HUN
							tag = PRC
							tag = CHI
							tag = JAP
							tag = TUR
							tag = FIN
							tag = RAJ
						}
					}
					set_variable = {
						YJ_country_limit = 30
					}
					set_variable = {
						YJ_country_cost = 75
					}
				}
				if = {
					limit = {
						OR = {
							tag = BEL
							tag = CZE
							tag = PHI
							tag = SWI
							tag = CAN
							tag = MEX
							tag = BRA
							tag = AUS
						}
					}
					set_variable = {
						YJ_country_limit = 50
					}
					set_variable = {
						YJ_country_cost = 50
					}
				}
				if = {
					limit = {
						NOT = {
							has_variable = YJ_country_limit
						}
					}
					set_variable = {
						YJ_country_limit = 50
					}
					set_variable = {
						YJ_country_cost = 30
					}
				}
			}
		}
	}

	on_daily = {
		effect = {
			every_country = {
				if = {
					limit = {
						has_country_flag = YJ_controller
						OR = {
							is_ai = yes
							surrender_progress > 0.75
						}
					}
					clr_country_flag = YJ_controller
				}

				if = {
					limit = {
						is_ai = no
						NOT = { has_idea = YJ_AIR_ideas }
					}
					add_ideas = YJ_AIR_ideas
					add_dynamic_modifier = {
						modifier = YJ_AIR_ideas
					}
				}
				if = {
					limit = {
						is_ai = yes
						has_idea = YJ_AIR_ideas
					}
					remove_ideas = YJ_AIR_ideas
					remove_dynamic_modifier = {
						modifier = YJ_AIR_ideas
					}
				}
			}

			if = {
				limit = {
					NOT = { any_country = { has_country_flag = YJ_controller } }
				}
				random_country = {
					limit = {
						is_ai = no
						surrender_progress < 0.75
					}
					set_country_flag = YJ_controller
				}
			}
		}
	}
}
